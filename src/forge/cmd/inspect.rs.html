22/// CLI arguments for `forge inspect`.
23#[derive(Clone, Debug, Parser)]
24pub struct InspectArgs {
25    /// The identifier of the contract to inspect in the form `(<path>:)?<contractname>`.
26    #[arg(value_parser = PathOrContractInfo::from_str)]
27    pub contract: PathOrContractInfo,
28
29    /// The contract artifact field to inspect.
30    #[arg(value_enum)]
31    pub field: ContractArtifactField,
32
33    /// All build arguments are supported
34    #[command(flatten)]
35    build: BuildOpts,
36
37    /// Whether to remove comments when inspecting `ir` and `irOptimized` artifact fields.
38    #[arg(long, short, help_heading = "Display options")]
39    pub strip_yul_comments: bool,
40}

42impl InspectArgs {
43    pub fn run(self) -> Result<()> {
44        let Self { contract, field, build, strip_yul_comments } = self;
45
46        trace!(target: "forge", ?field, ?contract, "running forge inspect");
47
48        // Map field to ContractOutputSelection
49        let mut cos = build.compiler.extra_output;
50        if !field.is_default() && !cos.iter().any(|selected| field == *selected) {
51            cos.push(field.into());
52        }
53
54        // Run Optimized?
55        let optimized = if field == ContractArtifactField::AssemblyOptimized {
56            Some(true)
57        } else {
58            build.compiler.optimize
59        };
60
61        // Build modified Args
62        let modified_build_args = BuildOpts {
63            compiler: CompilerOpts { extra_output: cos, optimize: optimized, ..build.compiler },
64            ..build
65        };
66
67        // Build the project
68        let project = modified_build_args.project()?;
69        let compiler = ProjectCompiler::new().quiet(true);
70        let target_path = find_target_path(&project, &contract)?;
71        let mut output = compiler.files([target_path.clone()]).compile(&project)?;
72
73        // Find the artifact
74        let artifact = find_matching_contract_artifact(&mut output, &target_path, contract.name())?;
75
76        // Match on ContractArtifactFields and pretty-print
77        match field {
78            ContractArtifactField::Abi => {
79                let abi = artifact
80                    .abi
81                    .as_ref()
82                    .ok_or_else(|| eyre::eyre!("Failed to fetch lossless ABI"))?;
83                print_abi(abi)?;
84            }
85            ContractArtifactField::Bytecode => {
86                print_json_str(&artifact.bytecode, Some("object"))?;
87            }
88            ContractArtifactField::DeployedBytecode => {
89                print_json_str(&artifact.deployed_bytecode, Some("object"))?;
90            }
91            ContractArtifactField::Assembly | ContractArtifactField::AssemblyOptimized => {
92                print_json_str(&artifact.assembly, None)?;
93            }
94            ContractArtifactField::LegacyAssembly => {
95                print_json_str(&artifact.legacy_assembly, None)?;
96            }
97            ContractArtifactField::MethodIdentifiers => {
98                print_method_identifiers(&artifact.method_identifiers)?;
99            }
100            ContractArtifactField::GasEstimates => {
101                print_json(&artifact.gas_estimates)?;
102            }
103            ContractArtifactField::StorageLayout => {
104                print_storage_layout(artifact.storage_layout.as_ref())?;
105            }
106            ContractArtifactField::DevDoc => {
107                print_json(&artifact.devdoc)?;
108            }
109            ContractArtifactField::Ir => {
110                print_yul(artifact.ir.as_deref(), strip_yul_comments)?;
111            }
112            ContractArtifactField::IrOptimized => {
113                print_yul(artifact.ir_optimized.as_deref(), strip_yul_comments)?;
114            }
115            ContractArtifactField::Metadata => {
116                print_json(&artifact.metadata)?;
117            }
118            ContractArtifactField::UserDoc => {
119                print_json(&artifact.userdoc)?;
120            }
121            ContractArtifactField::Ewasm => {
122                print_json_str(&artifact.ewasm, None)?;
123            }
124            ContractArtifactField::Errors => {
125                let out = artifact.abi.as_ref().map_or(Map::new(), parse_errors);
126                print_errors_events(&out, true)?;
127            }
128            ContractArtifactField::Events => {
129                let out = artifact.abi.as_ref().map_or(Map::new(), parse_events);
130                print_errors_events(&out, false)?;
131            }
132        };
133
134        Ok(())
135    }
136}

138fn parse_errors(abi: &JsonAbi) -> Map<String, Value> {
139    let mut out = serde_json::Map::new();
140    for er in abi.errors.iter().flat_map(|(_, errors)| errors) {
141        let types = get_ty_sig(&er.inputs);
142        let sig = format!("{:x}", er.selector());
143        let sig_trimmed = &sig[0..8];
144        out.insert(format!("{}({})", er.name, types), sig_trimmed.to_string().into());
145    }
146    out
147}
148
149fn parse_events(abi: &JsonAbi) -> Map<String, Value> {
150    let mut out = serde_json::Map::new();
151    for ev in abi.events.iter().flat_map(|(_, events)| events) {
152        let types = parse_event_params(&ev.inputs);
153        let topic = hex::encode(keccak256(ev.signature()));
154        out.insert(format!("{}({})", ev.name, types), format!("0x{topic}").into());
155    }
156    out
157}

159fn parse_event_params(ev_params: &[EventParam]) -> String {
160    ev_params
161        .iter()
162        .map(|p| {
163            if let Some(ty) = p.internal_type() {
164                return internal_ty(ty)
165            }
166            p.ty.clone()
167        })
168        .collect::<Vec<_>>()
169        .join(",")
170}

172fn print_abi(abi: &JsonAbi) -> Result<()> {
173    if shell::is_json() {
174        return print_json(abi)
175    }
176
177    let headers = vec![Cell::new("Type"), Cell::new("Signature"), Cell::new("Selector")];
178    print_table(headers, |table| {
179        // Print events
180        for ev in abi.events.iter().flat_map(|(_, events)| events) {
181            let types = parse_event_params(&ev.inputs);
182            let selector = ev.selector().to_string();
183            table.add_row(["event", &format!("{}({})", ev.name, types), &selector]);
184        }
185
186        // Print errors
187        for er in abi.errors.iter().flat_map(|(_, errors)| errors) {
188            let selector = er.selector().to_string();
189            table.add_row([
190                "error",
191                &format!("{}({})", er.name, get_ty_sig(&er.inputs)),
192                &selector,
193            ]);
194        }
195
196        // Print functions
197        for func in abi.functions.iter().flat_map(|(_, f)| f) {
198            let selector = func.selector().to_string();
199            let state_mut = func.state_mutability.as_json_str();
200            let func_sig = if !func.outputs.is_empty() {
201                format!(
202                    "{}({}) {state_mut} returns ({})",
203                    func.name,
204                    get_ty_sig(&func.inputs),
205                    get_ty_sig(&func.outputs)
206                )
207            } else {
208                format!("{}({}) {state_mut}", func.name, get_ty_sig(&func.inputs))
209            };
210            table.add_row(["function", &func_sig, &selector]);
211        }
212
213        if let Some(constructor) = abi.constructor() {
214            let state_mut = constructor.state_mutability.as_json_str();
215            table.add_row([
216                "constructor",
217                &format!("constructor({}) {state_mut}", get_ty_sig(&constructor.inputs)),
218                "",
219            ]);
220        }
221
222        if let Some(fallback) = &abi.fallback {
223            let state_mut = fallback.state_mutability.as_json_str();
224            table.add_row(["fallback", &format!("fallback() {state_mut}"), ""]);
225        }
226
227        if let Some(receive) = &abi.receive {
228            let state_mut = receive.state_mutability.as_json_str();
229            table.add_row(["receive", &format!("receive() {state_mut}"), ""]);
230        }
231    })
232}

234fn get_ty_sig(inputs: &[Param]) -> String {
235    inputs
236        .iter()
237        .map(|p| {
238            if let Some(ty) = p.internal_type() {
239                return internal_ty(ty);
240            }
241            p.ty.clone()
242        })
243        .collect::<Vec<_>>()
244        .join(",")

247fn internal_ty(ty: &InternalType) -> String {
248    let contract_ty =
249        |c: Option<&str>, ty: &String| c.map_or_else(|| ty.clone(), |c| format!("{c}.{ty}"));
250    match ty {
251        InternalType::AddressPayable(addr) => addr.clone(),
252        InternalType::Contract(contract) => contract.clone(),
253        InternalType::Enum { contract, ty } => contract_ty(contract.as_deref(), ty),
254        InternalType::Struct { contract, ty } => contract_ty(contract.as_deref(), ty),
255        InternalType::Other { contract, ty } => contract_ty(contract.as_deref(), ty),
256    }
257}

259pub fn print_storage_layout(storage_layout: Option<&StorageLayout>) -> Result<()> {
260    let Some(storage_layout) = storage_layout else {
261        eyre::bail!("Could not get storage layout");
262    };
263
264    if shell::is_json() {
265        return print_json(&storage_layout)
266    }
267
268    let headers = vec![
269        Cell::new("Name"),
270        Cell::new("Type"),
271        Cell::new("Slot"),
272        Cell::new("Offset"),
273        Cell::new("Bytes"),
274        Cell::new("Contract"),
275    ];
276
277    print_table(headers, |table| {
278        for slot in &storage_layout.storage {
279            let storage_type = storage_layout.types.get(&slot.storage_type);
280            table.add_row([
281                slot.label.as_str(),
282                storage_type.map_or("?", |t| &t.label),
283                &slot.slot,
284                &slot.offset.to_string(),
285                storage_type.map_or("?", |t| &t.number_of_bytes),
286                &slot.contract,
287            ]);
288        }
289    })
290}

292fn print_method_identifiers(method_identifiers: &Option<BTreeMap<String, String>>) -> Result<()> {
293    let Some(method_identifiers) = method_identifiers else {
294        eyre::bail!("Could not get method identifiers");

298        return print_json(method_identifiers)
299    }
300
301    let headers = vec![Cell::new("Method"), Cell::new("Identifier")];

303    print_table(headers, |table| {

310fn print_errors_events(map: &Map<String, Value>, is_err: bool) -> Result<()> {
311    if shell::is_json() {
312        return print_json(map);
313    }
314
315    let headers = if is_err {
316        vec![Cell::new("Error"), Cell::new("Selector")]
317    } else {
318        vec![Cell::new("Event"), Cell::new("Topic")]
319    };
320    print_table(headers, |table| {
321        for (method, selector) in map {
322            table.add_row([method, selector.as_str().unwrap()]);
323        }
324    })
325}

327fn print_table(headers: Vec<Cell>, add_rows: impl FnOnce(&mut Table)) -> Result<()> {
328    let mut table = Table::new();
329    table.apply_modifier(UTF8_ROUND_CORNERS);
330    table.set_header(headers);
331    add_rows(&mut table);
332    sh_println!("\n{table}\n")?;
333    Ok(())
334}

358macro_rules! impl_value_enum {
359    (enum $name:ident { $($field:ident => $main:literal $(| $alias:literal)*),+ $(,)? }) => {
360        impl $name {
361            /// All the variants of this enum.
362            pub const ALL: &'static [Self] = &[$(Self::$field),+];
363
364            /// Returns the string representation of `self`.
365            #[inline]
366            pub const fn as_str(&self) -> &'static str {
367                match self {
368                    $(
369                        Self::$field => $main,
370                    )+
371                }
372            }
373
374            /// Returns all the aliases of `self`.
375            #[inline]
376            pub const fn aliases(&self) -> &'static [&'static str] {
377                match self {
378                    $(
379                        Self::$field => &[$($alias),*],
380                    )+
381                }
382            }
383        }
384
385        impl ::clap::ValueEnum for $name {
386            #[inline]
387            fn value_variants<'a>() -> &'a [Self] {
388                Self::ALL
389            }
390
391            #[inline]
392            fn to_possible_value(&self) -> Option<::clap::builder::PossibleValue> {
393                Some(::clap::builder::PossibleValue::new(Self::as_str(self)).aliases(Self::aliases(self)))
394            }
395
396            #[inline]
397            fn from_str(input: &str, ignore_case: bool) -> Result<Self, String> {
398                let _ = ignore_case;
399                <Self as ::std::str::FromStr>::from_str(input)
400            }
401        }
402
403        impl ::std::str::FromStr for $name {
404            type Err = String;
405
406            fn from_str(s: &str) -> Result<Self, Self::Err> {
407                match s {
408                    $(
409                        $main $(| $alias)* => Ok(Self::$field),
410                    )+
411                    _ => Err(format!(concat!("Invalid ", stringify!($name), " value: {}"), s)),
412                }
413            }
414        }
415    };
416}

446impl From<ContractArtifactField> for ContractOutputSelection {
447    fn from(field: ContractArtifactField) -> Self {
448        type Caf = ContractArtifactField;
449        match field {
450            Caf::Abi => Self::Abi,
451            Caf::Bytecode => Self::Evm(EvmOutputSelection::ByteCode(BytecodeOutputSelection::All)),
452            Caf::DeployedBytecode => Self::Evm(EvmOutputSelection::DeployedByteCode(
453                DeployedBytecodeOutputSelection::All,
454            )),
455            Caf::Assembly | Caf::AssemblyOptimized => Self::Evm(EvmOutputSelection::Assembly),
456            Caf::LegacyAssembly => Self::Evm(EvmOutputSelection::LegacyAssembly),
457            Caf::MethodIdentifiers => Self::Evm(EvmOutputSelection::MethodIdentifiers),
458            Caf::GasEstimates => Self::Evm(EvmOutputSelection::GasEstimates),
459            Caf::StorageLayout => Self::StorageLayout,
460            Caf::DevDoc => Self::DevDoc,
461            Caf::Ir => Self::Ir,
462            Caf::IrOptimized => Self::IrOptimized,
463            Caf::Metadata => Self::Metadata,
464            Caf::UserDoc => Self::UserDoc,
465            Caf::Ewasm => Self::Ewasm(EwasmOutputSelection::All),
466            Caf::Errors => Self::Abi,
467            Caf::Events => Self::Abi,
468        }
469    }
470}

472impl PartialEq<ContractOutputSelection> for ContractArtifactField {
473    fn eq(&self, other: &ContractOutputSelection) -> bool {
474        type Cos = ContractOutputSelection;
475        type Eos = EvmOutputSelection;
476        matches!(
477            (self, other),
478            (Self::Abi | Self::Events, Cos::Abi) |
479                (Self::Errors, Cos::Abi) |
480                (Self::Bytecode, Cos::Evm(Eos::ByteCode(_))) |
481                (Self::DeployedBytecode, Cos::Evm(Eos::DeployedByteCode(_))) |
482                (Self::Assembly | Self::AssemblyOptimized, Cos::Evm(Eos::Assembly)) |
483                (Self::LegacyAssembly, Cos::Evm(Eos::LegacyAssembly)) |
484                (Self::MethodIdentifiers, Cos::Evm(Eos::MethodIdentifiers)) |
485                (Self::GasEstimates, Cos::Evm(Eos::GasEstimates)) |
486                (Self::StorageLayout, Cos::StorageLayout) |
487                (Self::DevDoc, Cos::DevDoc) |
488                (Self::Ir, Cos::Ir) |
489                (Self::IrOptimized, Cos::IrOptimized) |
490                (Self::Metadata, Cos::Metadata) |
491                (Self::UserDoc, Cos::UserDoc) |
492                (Self::Ewasm, Cos::Ewasm(_))
493        )
494    }

520fn print_yul(yul: Option<&str>, strip_comments: bool) -> Result<()> {
521    let Some(yul) = yul else {
522        eyre::bail!("Could not get IR output");
523    };
524
525    static YUL_COMMENTS: LazyLock<Regex> =
526        LazyLock::new(|| Regex::new(r"(///.*\n\s*)|(\s*/\*\*.*?\*/)").unwrap());
527
528    if strip_comments {
529        sh_println!("{}", YUL_COMMENTS.replace_all(yul, ""))?;
530    } else {
531        sh_println!("{yul}")?;
532    }
533
534    Ok(())
535}

537fn get_json_str(obj: &impl serde::Serialize, key: Option<&str>) -> Result<String> {
538    let value = serde_json::to_value(obj)?;
539    let mut value_ref = &value;
540    if let Some(key) = key {
541        if let Some(value2) = value.get(key) {
542            value_ref = value2;
543        }
544    }
545    let s = match value_ref.as_str() {
546        Some(s) => s.to_string(),
547        None => format!("{value_ref:#}"),
548    };
549    Ok(s)
550}
